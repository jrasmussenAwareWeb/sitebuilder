Code Sample

<div class="sidenav" style="">
    <ul class="nav nav-pills nav-stacked">
        <li class="active" role="presentation">
            <!-- parent -->
            <a href="/Components/Content">Content</a>
            <!-- end parent -->
            <ul class="nav">
                <!-- sibling page -->
                <li>
                    <a href="/Components/Content/Accordion">Accordion</a>
                </li>
            </ul>
        </li>
    </ul>
</div>
